公式推导

回到我们需要求的公式，现在我们要开始推公式啦！

提前注明：之后的公式为了方便，统一省略了 。

第二行到第三行的转化可以有两种方法理解，第一种就是非常直观的列举前几项感受一下具体$ \mu^2(i)$ 会被计算几次。第二种是利用二重积分交换积分限的思路来理解，事实上，第一种方法更加的直观。
之后我们要集中处理 $ \sum_{i=1}^{n} \mu^2(i)*i^2 $，这里使用到了容斥的思想，我会对这一部分进详细的分析。

首先我们来思考一道更加简单的题目bzoj2440，这个题目需要解决的一个子问题是，给定区间$[1,n]$求出这个区间中不含有平方因子的数的个数。本质上是要求解$\sum_{i=1}^{n}\mu^2(i)$。

事实上这个公式可以利用容斥原来的思想 推导出更加方便运算的形式：

大家可以跟着我一起来思考，对于公式左边的$i$，只有当$i$不含有平方因子时，$\mu(i)$不等于$0$，对于最后的结果才有贡献。

第二步就是要利用容斥来计算出答案，对于一个平方因子$i^2$，$[1,n]$中含有该平方因子的数的个数是$\lfloor\frac{n}{i^2}\rfloor $，

设最后的答案为$ans$，$ans=n-含有2^2因子的个数-含有3^2因子的个数+含有2^2*3^2因子的个数…$，容斥的系数刚好就是$\mu(i)$。

我们可以通过分析前几项的方式来感受一下：

当$i=1$时，$\lfloor\frac{n}{1^2}\rfloor $代表这个区间所有数的数量，这一项的系数为1；

当$i=2$时，含有平方因子$2^2$的数的数量是$\lfloor\frac{n}{2^2}\rfloor$，我们需要减去这个数，这一项的系数为-1；

当$i=3$时，含有平方因子$3^2$的数的数量是$\lfloor\frac{n}{3^2}\rfloor$，我们需要减去这个数，这一项的系数为-1；

当$i=4$时，含有平方因子$4^2$的数的数量是$\lfloor\frac{n}{4^2}\rfloor$，因为在$i=2$时我们已经处理了含有平方因子$4^2$的数，所以这一项的系数是0。

当$i=5$时，含有平方因子$5^2$的数的数量是$\lfloor\frac{n}{5^2}\rfloor$，我们需要加上这个数，这一项的系数为1；

当$i=6$时，含有平方因子$6^2$的数的数量是$\lfloor\frac{n}{6^2}\rfloor$，在$i=2$和$i=3$时，含有平方因子$6^2$的数被减了两次，所以我们需要加上这个数，这一项的系数为1；

当$i=7$时，含有平方因子$7^2$的数的数量是$\lfloor\frac{n}{7^2}\rfloor$，我们需要加上这个数，这一项的系数为1；

当$i=8$时，含有平方因子$8^2$的数的数量是$\lfloor\frac{n}{7^2}\rfloor$，在$i=2$时，含有平方因子$8^2$的数已经被处理过了，所以这一项的系数为0；

……

通过前几项的分析，相信大家对于容斥以及枚举平方因子时系数的选择的过程一定有了一个直观的理解。

int128的使用

运算过程种会爆long long，所以要使用int128，int218的参考资料 [5]

下面是使用int128的模板，实际上这个题目当中int128是作为一个中间变量来使用的。

#include <bits/stdc++.h>
using namespace std;


inline __int128 read()
{
    __int128 x=0,f=1;
    char ch=getchar();
    while(ch<'0'||ch>'9')
    {
        if(ch=='-')
            f=-1;
        ch=getchar();
    }
    while(ch>='0'&&ch<='9')
    {
        x=x*10+ch-'0';
        ch=getchar();
    }
    return x*f;
}

inline void write(__int128 x)
{
    if(x<0)
    {
        putchar('-');
        x=-x;
    }
    if(x>9)
        write(x/10);
    putchar(x%10+'0');
}

int main()
{
    __int128 a = read();
    __int128 b = read();
    write(a + b);
    return 0;
}

#include <cstdio>
#include <algorithm>
#include <cstring>
#include <iostream>
#include <string>
#include <queue>
#include <set>
#include <map>
#include <vector>
#include <cmath>
using namespace std;
#define inc(i,a,b) for(int i=a;i<=b;i++)
#define dec(i,a,b) for(int i=a;i>=b;i--)
#define ll long long 
#define mst(a,b) memset(a,b,sizeof a)
#define pb push_back
#define sqr(x) ((x)*(x))

const int maxv=1e5+10;
bool np[maxv];
int mu[maxv],prime[maxv],tot;

void sieve(int n){
	tot=0;mst(mu,0);mst(np,0);
	mu[1]=1;np[1]=true;
	inc(i,2,n){
		if(!np[i]) prime[++tot]=i,mu[i]=-1;
		inc(j,1,tot){
			if(i*prime[j]>n) break;
			np[i*prime[j]]=1;
			if(i%prime[j]==0){
			   mu[i*prime[j]]=0;
			   break;
			}
			mu[i*prime[j]]=mu[i]*mu[prime[j]];
		}
	}
}

__int128 ans,p,inf;
ll n; 
ll sum(ll n){
	__int128 x=n,y=n+1,z=2*n+1;
	if(x%2==0) x/=2;else if(y%2==0) y/=2;else z/=2;
	if(x%3==0) x/=3;else if(y%3==0) y/=3;else z/=3;
	return (ll)(x*y%p*z%p);
}

ll pre(ll n){
	__int128 x=n,y=n+1;
	if(x%2==0) x/=2;else y/=2;
	x=x*y%p;x=x*x%p;return x;
}

int main(){	

	ll o;
	sieve(100000);
	while(~scanf("%lld%lld",&n,&o)){
		p=o;
		ans=0;
		for(ll i=1;i*i<=n;i++){
			__int128 ff=1LL*mu[i]*i*i*i%p*i%p*i%p*i%p;
			__int128 gg=1LL*mu[i]*i*i%p*i%p*i%p;
			ans+=p+gg*sum(n/sqr(i))%p*(n+1)%p-ff*pre(n/sqr(i))%p;
			ans%=p;
		}
		printf("%lld\n",(ll)ans);
	}
	return 0;
}

java版本代码

写完之后我还写了一个java版本的代码，因为题目没有提示java的时间限制是多少，估计出题方开的时间限制和c++版本的是一样的，写完之后提交果然T飞了（时间超限），所以写了一个生成随机数的程序，简单对拍了一下。下面的java 代码应该是没有问题的。

使用java编写代码，其实就是为了使用BigInteger类（大整数类），这样可以使用无限范围大的整数。

BigInteger大整数类的参考资料 Java中的BigInteger在ACM中的应用[6]。

java编写程序时必须是要使用Main作为类名。

// 1500ms 的限制下tle了
// 对java这么不友好吗
// 对拍了应该是对的
//
import java.math.BigInteger;
import java.util.Scanner;

public class Main {
    static int maxv=(int)(1e5+10);
    static int[] mu=new int[maxv];
    static int[] prime=new int[maxv];
    static boolean[] np=new boolean[maxv];
    static int tot;
    static BigInteger ans= new BigInteger("0");
    static BigInteger p = new BigInteger("0");
    static BigInteger n = new BigInteger("0");

    static void sieve(int n){
        tot=0;
        for(int i=0;i<maxv;i++) {
            mu[i] = 0;
            np[i] = false;
        }
        mu[1]=1;
        np[1]=true;
        for(int i=2;i<=n;i++){
            if(!np[i]){
                prime[++tot]=i;
                mu[i]=-1;
            }
            for(int j=1;j<=tot;j++){
                if(i*prime[j]>n) break;
                np[i*prime[j]]=true;
                if(i%prime[j]==0){
                    mu[i*prime[j]]=0;
                    break;
                }
                mu[i*prime[j]]=mu[i]*mu[prime[j]];
            }
        }
    }

    static BigInteger sum(BigInteger n){
        BigInteger x=n;
        BigInteger y=n.add(BigInteger.ONE);
        BigInteger z=n.multiply(BigInteger.valueOf(2)).add(BigInteger.valueOf(1));
        if(x.mod(BigInteger.valueOf(2)).equals(BigInteger.ZERO)) x=x.divide(BigInteger.valueOf(2));
        else if(y.mod(BigInteger.valueOf(2)).equals(BigInteger.ZERO)) y=y.divide(BigInteger.valueOf(2));
        else z=z.divide(BigInteger.valueOf(2));
        if(x.mod(BigInteger.valueOf(3)).equals(BigInteger.ZERO)) x=x.divide(BigInteger.valueOf(3));
        else if(y.mod(BigInteger.valueOf(3)).equals(BigInteger.ZERO)) y=y.divide(BigInteger.valueOf(3));
        else z=z.divide(BigInteger.valueOf(3));
        return x.multiply(y).mod(p).multiply(z).mod(p);
    }

    static BigInteger pre(BigInteger n){
        BigInteger x=n;
        BigInteger y=n.add(BigInteger.ONE);
        if(x.mod(BigInteger.valueOf(2)).equals(BigInteger.ZERO)) x=x.divide(BigInteger.valueOf(2));
        else y=y.divide(BigInteger.valueOf(2));
        x=x.multiply(y).mod(p);
        x=x.multiply(x).mod(p);
        return x;
    }
    static public void main(String[] args){
        sieve(100000);
        Scanner in = new Scanner(System.in);
        while(in.hasNext()){
            n=in.nextBigInteger();
            p=in.nextBigInteger();
            ans=BigInteger.ZERO;
            BigInteger i=BigInteger.valueOf(1);
            for(;i.multiply(i).compareTo(n)<=0;i=i.add(BigInteger.ONE)){
                int ip=i.intValue();
                BigInteger ff=BigInteger.valueOf(mu[ip]);
                ff=ff.multiply(i).multiply(i).multiply(i).mod(p).multiply(i).mod(p).multiply(i).mod(p).multiply(i).mod(p);
                BigInteger gg=BigInteger.valueOf(mu[ip]);
                gg=gg.multiply(i).multiply(i).mod(p).multiply(i).mod(p).multiply(i).mod(p);
                ans=ans.add(p);
                BigInteger fi=sum(n.divide(i.multiply(i)));
                fi=fi.multiply(gg).mod(p).multiply(n.add(BigInteger.ONE)).mod(p);
                BigInteger se=pre(n.divide(i.multiply(i))).multiply(ff).mod(p);
				ans=ans.add(fi).subtract(se).mod(p);
            }
            System.out.println(ans);
        }
    }
}

	else

		cat data.in

	fi
done
