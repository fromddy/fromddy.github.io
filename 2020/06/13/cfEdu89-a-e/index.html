cfEdu89的A题-E题

首先放题目链接 Educational Codeforces Round 89

下面是A-E题目我的思路总结。

A题

题意归纳

制作某种工具有两个方式，第一种是1个材料A+2个材料B，第二种是2个材料A+1个材料B，现在有A材料$\ a\ $个，B材料$\ b\ $个，问最多能制作几个想要的工具。

输入数据

第一行输入数据组数 $t$, 数据范围$1\leq t \leq 1000$。

之后$t$行，每行有两个整数 $a$ 和 $b$ ，$(0\leq a,b\leq 10^9)$。

核心思路

思维题。

如果$a>b$的话就先调换顺序,保证$a<b$。

得到一个差值$cnt=b-a$，如果这个差值刚好大于等于$a$，那么最后的答案等于$a$；

如果$cnt<a$，那么我们需要处理的子问题就变成了材料A和材料B的数量均为$a-cnt$时能造多少工具，这里具体细节见代码吧。

B题

一句话题解

不停地增加含有x的区间的交集。

代码

//省略了代码头部，可以参考A题目的代码头部
int t;
int n,x,m;
const int maxv=1e2+10;
int l[maxv],r[maxv];

int main(){
	t=read();while(t--){
		n=read();
		x=read();
		m=read();
		int tl=x,tr=x;
		inc(i,1,m){
			int l=read();
			int r=read();
			//两个区间有交集，就可以选择扩大当前选择的区间了
			if(tl>=l&&tl<=r) tl=min(tl,l),tr=max(tr,r);
			else if(tr>=l&&tr<=r) tl=min(tl,l),tr=max(tr,r);
			else if(l>=tl&&l<=tr) tl=min(tl,l),tr=max(tr,r);
			else if(r>=tl&&r<=tr) tl=min(tl,l),tr=max(tr,r);
		}
		printf("%d\n",tr-tl+1);
	}
	return 0;
}

C题

一句话题解

要求一个矩形矩阵从$(1,1)$到$(n,m)$的路径上的0或1是对称的，

那么对于每一个坐标我们可以求解一个值，

$dis=min(该点到(1,1)的距离，该点到(n,m)的距离)$

这个值相同的点上的数必须是相同的。

特别注意的是 到$(1,1)$和$(n,m)$的距离相同的点不需要纳入计算考虑范围。

代码

//省略了头文件
int t;
const int maxv=30+5;
int a[maxv][maxv];
int n,m;

map<int,int> zero;
map<int,int> one;

int main(){
	t=read();while(t--){
		n=read();
		m=read();
		inc(i,1,n) inc(j,1,m) scanf("%d",&a[i][j]);
		zero.clear();
		one.clear();

		int sum=n+m-2;

		int cnt=0;
		inc(i,1,n) inc(j,1,m){
			int now=min(i-1+j-1,sum-(i-1+j-1));
			if(i-1+j-1==sum-(i-1+j-1)) continue;
			cnt=max(cnt,now);
			if(a[i][j]==0) zero[now]++;
			else if(a[i][j]==1) one[now]++;
		}

		int ans=0;
		inc(i,0,cnt){

			ans+=min(zero[i],one[i]);
		}
		printf("%d\n",ans);


	}
	return 0;
}

D题目

题意

给你$n$个数$a_1,a_2,a_3,…,a_n$。

对于每一个$a_i$ 需要找到它的两个因子$d_1>1$和$d_2>1$满足$gcd(d_1+d_2,a_i)=1$，如果满足的话那么就输出 $d_1\ d_2$ ，如果 找不到就输出$-1 \ -1$。

思路

这种题当然要往分解质因数的角度去考虑啦，我们知道任何的一个数，都可以表示成质因子乘积的形式，如下所示：

我们令$d_1=p_1^{n_1}$，$d_2=a/d_1$，我们可以惊喜地发现是满足条件的。

其实我也想了很久的啊=_=，但是结论的确非常简单。

//省略了头文件
const int maxv=1e7+10;
const int maxn=5e5+10;
int prime[maxv];
int xiao[maxv];
int tot;
bool np[maxv];

void sieve(){
	int Max=1e7;
	inc(i,0,Max) np[i]=false;

	tot=0; np[1]=true;
	for(int i=2;i<=Max;i++){
		if(np[i]==false){
			prime[++tot]=i;
			xiao[i]=i;
	    //	cout<<i<<" "<<xiao[i]<<endl;
		}

		for(int j=1;j<=tot;j++){
			if(i*prime[j]>Max) break;
			np[i*prime[j]]=true;
			///cout<<i*prime[j]<<" "<<prime[j]<<endl;

			xiao[i*prime[j]]=prime[j];
			if(i%prime[j]==0) break;
		}
	}
}

int l[maxn],r[maxn];
int main(){
	sieve();
	n=read();

		int a=read();

		else {

#define rt o
#define ls o<<1
#define rs o<<1|1
#define lson l,m,ls
#define rson m+1,r,rs
const int maxv=2e5+10;
int a[maxv];
int b[maxv];
int Min[maxv<<2];

int n,m;
int cur;

void pushup(int rt){
	Min[rt]=min(Min[ls],Min[rs]);
}

void build(int l,int r,int rt){
	if(l==r){
		Min[rt]=a[++cur];
		return;
	}
	int m=(l+r)>>1;
	build(lson);
	build(rson);
	pushup(rt);
	return ;
}

void update(int pos,int num,int l,int r,int rt){
	if(l==r){
		Min[rt]=num;
		return ;
	}
	int m=(l+r)>>1;
	if(pos<=m) update(pos,num,lson);
	else update(pos,num,rson);
	pushup(rt);
}

int query(int ql,int qr,int l,int r,int rt){
	if(ql<=l&&r<=qr) return Min[rt];
	int m=(l+r)>>1;
	int _=INT_MAX;
	if(ql<=m) _=min(_,query(ql,qr,lson));
	if(qr>m) _=min(_,query(ql,qr,rson));
	return _;
}


map<int,int> mp;
ll mod=998244353; 
bool show=true;


int main(){
	show=false;
	while(~scanf("%d%d",&n,&m)){
		mp.clear();
		inc(i,1,n) {
			scanf("%d",a+i);

			mp[a[i]]=i;
		}
		inc(i,1,m) scanf("%d",b+i);
		cur=0;
		build(1,n,1);
		bool flag=true;
		ll ans=0;
		inc(i,1,m) if(mp[b[i]]==0) flag=false;
		if(flag){
		
			ans=1;
		if(query(1,mp[b[1]],1,n,1)!=b[1]) flag=false;
		if(query(mp[b[m]],n,1,n,1)!=b[m]) flag=false;

	for(int i=1;i<=m-1;i++){
			int l=mp[b[i]];
			int r=mp[b[i+1]];

			int cnt=0;
			for(int j=r;j>l;j--){
				int fi=query(l,j-1,1,n,1);
				int se=query(j,r,1,n,1);
				if(fi==b[i]&&se==b[i+1]) cnt++;
				else break;
			}
			ans=(ans*cnt)%mod;
		}
		}

		if(flag==false) ans=0;
		printf("%lld\n",ans);
	}
	return 0;
}
