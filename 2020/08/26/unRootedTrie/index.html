#include <bits/stdc++.h>
using namespace std;
#define inc(i,a,b) for(int i=a;i<=b;i++)
#define dec(i,a,b) for(int i=a;i>=b;i--)
#define ll long long 
#define mst(a,b) memset(a,b,sizeof a)
#define pb push_back
inline int read(){
	char ch=getchar();int f=1,x=0;
	while(!isdigit(ch)) { if(ch=='-') f=-1;ch=getchar();}
	while(isdigit(ch)){ x=x*10+ch-'0';ch=getchar();}
	return f*x;
}
int n,m;
const int maxv=1e5+10;
struct Edge{ int to,nxt;char ch; } edge[maxv*2];
int head[maxv]; int top;

void addedge(int from,int to,char ch){
	edge[top].to=to;
	edge[top].ch=ch;
	edge[top].nxt=head[from];
	head[from]=top++;
}

int pos[maxv],sz[maxv],f[maxv];
char fColor[maxv];
int cnt;
void dfs(int u,int fa){
	pos[u]=++cnt;
	sz[u]=1;
	f[u]=fa;
	fColor[u]=0;

	for(int i=head[u];~i;i=edge[i].nxt){
		int to=edge[i].to;
		char ch=edge[i].ch;
		if(to==fa) {
			fColor[u]=ch;
			continue;
		}
		dfs(to,u);
		sz[u]+=sz[to];
	}
}

int hav[maxv][30];
int twoColor[maxv];
int val[maxv];

void update(int l,int r,int o){
	val[l]+=o;
	val[r+1]-=o;
	return;
}

int main(){
	int T=read();while(T--){
		cnt=top=0;
		n=read();
		inc(i,1,n) {
			head[i]=-1;
			inc(j,0,30) hav[i][j]=0;
			val[i]=0;
			twoColor[i]=-1;
		}

		inc(i,1,n-1){
			int u=read();
			int v=read();
			char ch; scanf(" %c",&ch);
			addedge(u,v,ch);
			addedge(v,u,ch);
		}

		bool ok=true;

		for(int i=1;i<=n;i++){
			for(int j=head[i];~j;j=edge[j].nxt){
				char ch=edge[j].ch;
				hav[i][ch-'a']++;
			}
			int cnt2=0,cnt3=0;
			for(char ch='a';ch<='z';ch++){
				if(hav[i][ch-'a']==2) twoColor[i]=ch,cnt2++;
				else if(hav[i][ch-'a']>=3) cnt3++;
			}
			if(cnt3>=1) ok=false;
			if(cnt2>1) ok=false; 
		}
		if(ok==false) { printf("0\n"); continue; }
		dfs(1,-1);
		for(int i=1;i<=n;i++){
			if(twoColor[i]==-1) continue;
			update(pos[i],pos[i],-1);
			int tc=twoColor[i];
			int fc=fColor[i];
			if(fc==-1){
				for(int j=head[i];~j;j=edge[j].nxt){
					int to=edge[j].to;
					char ch=edge[j].ch;
					if(to==f[i]) continue;
					if(ch!=tc){
						int l=pos[to];
						int r=pos[to]+sz[to]-1;
						update(l,r,-1);
					}
				}
			}
			else if(fc==tc){
				for(int j=head[i];~j;j=edge[j].nxt){
					int to=edge[j].to;
					char ch=edge[j].ch;
					if(to==f[i]) continue;
					if(ch==fc) continue;
					int l=pos[to];
					int r=pos[to]+sz[to]-1;
					update(l,r,-1);
				}
			}else {
				update(1,n,-1);
				for(int j=head[i];~j;j=edge[j].nxt){
					int to=edge[j].to;
					char ch=edge[j].ch;
					if(to==f[i]) continue;
					if(ch==tc) {
						int l=pos[to];
						int r=pos[to]+sz[to]-1;
						update(l,r,1);
					}
				}
			}
		}
		for(int i=1;i<=n+1;i++) val[i]+=val[i-1];
		int ans=0;
		for(int i=1;i<=n;i++) if(val[i]>=0) ans++;
		printf("%d\n",ans);
	}
	return 0;

}
