1001 Total Eclipse

题目大意

给定一个$n$个点$m$条边的图，每个点有一个亮度值 $b_i$，最终的目标是使得所有点的亮度值都变成$0$。

每一次的操作可以选择 $k$ 个不同的点，这些点需要是联通的，对于这些点亮度值全部都减去1。

需要注意的是每一次总会选择数量最大的联通快。

输入

首先是数据组数 $T（1 \le T \le 10)$ ，

对于每组数据，输入 边数 $n (1 \le n \le 100000 )$ 点数 $m(1 \e m \e 200000) ，

之后输入 n个灯的亮度$b_1,b_2,…,b_n ( 1 \le b_i \le 10^9)$。

之后的 $m$ 行，输入$ u_i $和$ v_i (1 \le u_i,v_i \le n ,u_i \ne v_i )$ ，表示双向边，注意这里在两个城市之间可能有重边。

输出

对于每一组数据，输出最小的操作数。

思路

首先在脑海里构建这样一个情景，n个点在三维空间当中代表n个柱子，高度就是给定的 b_i ，然后水逐渐升高漫过了所有的柱子，利用逆向思维，想到所有水从最高处逐渐下降，去统计最后的答案。

最后确定的思路： 首先把所有的亮度全部加到$ans$上，之后模拟水逐渐下降的过程，具体就是把所有的边按照两个点最小的高度排序，之后倒着加边。如果这两个点不属于同一个集合，那么就将这两个点连接起来，如果这两个点属于同一个集合，那么这两个点当中比较低的高度其实是重复计算的，那么在$ans$当中减去$min(d[u_i],d[v_i]) $。

判断点属不属于同一个集合需要使用到并查集。

最后答案可能比较大，所以$ans$记得开$long long$。

#include <bits/stdc++.h>
using namespace std;
#define inc(i,a,b) for(int i=a;i<=b;i++)
#define dec(i,a,b) for(int i=a;i>=b;i--)
#define ll long long 
#define mst(a,b) memset(a,b,sizeof a)
#define pb push_back
inline int read(){
    char ch=getchar();int f=1,x=0;
	while(!isdigit(ch)) { if(ch=='-') f=-1;ch=getchar();}
	while(isdigit(ch)){ x=x*10+ch-'0';ch=getchar();}
	return f*x;
}

const int maxv=1e5+10;
int par[maxv];
int h[maxv];

void init(int n){
	inc(i,0,n){
		par[i]=i;
		h[i]=0;
	}
}

int find(int x){
	if(x==par[x]) return x;
	return par[x]=find(par[x]);
}


void unite(int x,int y){
	x=find(x),y=find(y);
	if(x==y) return;
	if(h[x]>h[y]) par[y]=x;
	else {
		par[x]=y;
		if(h[x]==h[y]) h[y]++;
	}
}

bool same(int x,int y){
	return find(x)==find(y);
}



int d[maxv];

int n,m;


const int maxe=2e5+10;

struct bian{
	int u,v;

	bool operator<(const bian & rhs) const{
		int fi=min(d[u],d[v]);
		int se=min(d[rhs.u],d[rhs.v]);
		return fi<se;
	}

}bian[maxe];


int main(){
	int T=read();while(T--){
		int n=read();
		int m=read();
		ll ans=0;
		inc(i,1,n){
		   	scanf("%d",d+i);
			ans+=1LL*d[i];
			par[i]=i;
			h[i]=0;
		}

		inc(i,1,m){
			int u=read();
			int v=read();
			bian[i].u=u;
			bian[i].v=v;
		}
		sort(bian+1,bian+1+m);

		for(int i=m;i>=1;i--){
			int u=bian[i].u;
			int v=bian[i].v;
			ll temp=min(d[u],d[v]);

			if(find(u)!=find(v)){
//				cout<<u<<" "<<v<<endl;
			   	ans-=temp;
				unite(u,v);
			}
		}
		printf("%lld\n",ans);
	}
	return 0;
}
